// Login Component with NAuth
import { useState } from 'react'
import { IUserProvider, UserContext } from "@/lib/nauth-core";

export function LoginForm() {

  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');

  const userContext = useContext<IUserProvider>(UserContext);

  const handleSubmit = async (e) => {
    e.preventDefault()
    let ret = await userContext.loginWithEmail(email, password); 
    if (!ret.sucesso) {
      setError(ret.mensagemErro);
      return;
    }
    // Redirect to dashboard page
    window.location.href = '/dashboard';
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <div>
        <label htmlFor="email">Email</label>
        <input
          id="email"
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          required
        />
      </div>
      <div>
        <label htmlFor="password">Password</label>
        <input
          id="password"
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          required
        />
      </div>
      <button type="submit" disabled={loading}>
        {userContext.loading ? 'Signing in...' : 'Sign In'}
      </button>
      {error && <p className="error">{error}</p>}
    </form>
  )
}
